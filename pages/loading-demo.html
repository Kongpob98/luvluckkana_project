วิธีการใช้งาน

1. เพิ่ม CSS และ JS ในหน้า HTML

<link rel="stylesheet" href="../css/loading.css">
<script src="../js/components/loading.js"></script>

2. Loading จะแสดงอัตโนมัติเมื่อโหลดหน้า

Component จะแสดงอัตโนมัติและซ่อนเมื่อหน้าโหลดเสร็จ

3. ควบคุมด้วย JavaScript (Optional)

// แสดง loading
window.loadingScreen.show();

// ซ่อน loading
window.loadingScreen.hide();

// สร้าง loading ใหม่ด้วย options
const customLoading = LoadingScreen.init({
    videoSrc: 'https://res.cloudinary.com/dpezsckqq/video/upload/luckkana/loadd.mp4',
    minDisplayTime: 2000, // แสดงอย่างน้อย 2 วินาที
    showProgress: true, // แสดงข้อความ "กำลังโหลด"
    autoHide: true // ซ่อนอัตโนมัติเมื่อหน้าโหลดเสร็จ
});

4. ใช้กับการโหลดข้อมูล

// แสดง loading ก่อนโหลดข้อมูล
window.loadingScreen.show();

// โหลดข้อมูล
fetch('/api/data')
    .then(response => response.json())
    .then(data => {
        // ทำอะไรกับข้อมูล
        console.log(data);
    })
    .finally(() => {
        // ซ่อน loading เมื่อเสร็จ
        window.loadingScreen.hide();
    });

Options Configuration

{
    videoSrc: 'https://res.cloudinary.com/dpezsckqq/video/upload/luckkana/loadd.mp4',  // ที่อยู่วีดีโอ
    minDisplayTime: 1500,                     // เวลาแสดงขั้นต่ำ (ms)
    showProgress: true,                       // แสดงข้อความ "กำลังโหลด"
    autoHide: true                            // ซ่อนอัตโนมัติเมื่อโหลดเสร็จ
}
